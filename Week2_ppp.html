    c = trial.suggest_float("C", 0.001, 1.0)

    model = LogisticRegression(
        class_weight='balanced',
        C=c,
        random_state=random_state
    )
    
    scores = cross_val_score(
        model, 
        X=X_train_scaled, 
        y=y_train, 
        cv=cv,
        scoring='recall',
        n_jobs=-1,
    )
    recall = scores.mean()
    
    return recall
